{{- $page := . -}} {{- $src := "" -}} {{- if $page.IsHome -}} {{- $src = "images/home-og.jpg" -}} {{- else -}} {{- $src = $page.Params.og_image | default $page.Params.image | default $page.Params.featured_image | default "images/home-og.jpg" -}} {{- end -}} {{- $fallbackSrc := "images/home-og.jpg" -}} {{- $fallback := partial "image-variants.html" (dict "src" $fallbackSrc) -}} {{- $result := $fallback -}} {{- $img := false -}} {{- if $src -}} {{- if or (hasPrefix $src "http://") (hasPrefix $src "https://") -}} {{- with try (resources.GetRemote $src) -}} {{- with .Err -}} {{- warnf "Không tải được ảnh để tạo OG card, dùng fallback: %s" $src -}} {{- else with .Value -}} {{- $img = . -}} {{- end -}} {{- end -}} {{- else -}} {{- $local := strings.TrimPrefix "/" $src -}} {{- with resources.Get $local -}} {{- $img = . -}} {{- else -}} {{- warnf "Không tìm thấy ảnh local để tạo OG card, dùng fallback: %s" $src -}} {{- end -}} {{- end -}} {{- end -}} {{- if not $img -}} {{- with resources.Get $fallbackSrc -}} {{- $img = . -}} {{- end -}} {{- end -}} {{- with $img -}} {{/* OG v3 Mobile-safe layout: - Không overlay title nhỏ lên ảnh nữa. - Facebook mobile đã có title riêng bên dưới thumbnail. - Ảnh tattoo là trọng tâm: rõ, to, ít bị crop. - Dùng blur background để ảnh không bị viền trống khi contain. */}} {{- $bg := .Fill "1200x630 Center jpg q82" -}} {{- $bg = $bg.Filter (images.GaussianBlur 24) (images.Brightness -38) -}} {{- $card := $bg -}} {{/* Ảnh chính: fit để giữ trọn hình nhiều nhất có thể */}} {{- $piece := .Fit "1060x570 jpg q95" -}} {{- $pieceX := div (sub 1200 $piece.Width) 2 -}} {{- $pieceY := div (sub 630 $piece.Height) 2 -}} {{- $card = $card.Filter (images.Overlay $piece $pieceX $pieceY) -}} {{/* Overlay nhận diện thương hiệu nhẹ, không phá ảnh */}} {{- with resources.Get "images/og/lanpink-og-v3-frame.png" -}} {{- $card = $card.Filter (images.Overlay . 0 0) -}} {{- end -}} {{- $result = dict "original" (dict "src" .RelPermalink "abs" .Permalink "width" .Width "height" .Height "type" "image/jpeg") "hero" (dict "src" $card.RelPermalink "abs" $card.Permalink "width" $card.Width "height" $card.Height "type" "image/jpeg") "og" (dict "src" $card.RelPermalink "abs" $card.Permalink "width" $card.Width "height" $card.Height "type" "image/jpeg") "thumb" (dict "src" $piece.RelPermalink "abs" $piece.Permalink "width" $piece.Width "height" $piece.Height "type" "image/jpeg") -}} {{- end -}} {{- return $result -}}
